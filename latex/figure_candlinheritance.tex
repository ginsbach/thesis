\begin{lstlisting}[language=CAnDL]
Constraint Sum
( opcode{out} = add
([$\tt \land$]) {in1} = {out}.args[0]
([$\tt \land$]) {in2} = {out}.args[1])
End
Constraint SumOfSums
( include Sum@{sum1}
([$\tt \land$]) include Sum@{sum2}
([$\tt \land$]) include Sum({sum1.out}->{in1},{sum2.out}->{in2}))
End
\end{lstlisting}
